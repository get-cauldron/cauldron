```typescript
import type { Result } from '@get-cauldron/types'
import { err } from '@get-cauldron/types'

export interface CauldronConfig {
	name: string
	extensions?: string[]
}

export async function loadConfig(projectDir: string): Promise<Result<CauldronConfig>> {
	try {
		const configPath = join(projectDir, '.cauldron', 'config.json')
		const content = await readFile(configPath, 'utf8')
		const config = JSON.parse(content) as CauldronConfig
		return { ok: true, value: config }
	} catch (cause) {
		return err(new Error('Cannot read config', { cause }))
	}
}
```

```typescript
/**
 * Cauldron configuration schema and types.
 *
 * Defines the validated config shape via Zod. CauldronConfig type is derived
 * from the schema so the runtime validation and the TypeScript type stay in sync.
 */

import * as z from 'zod'

export const CauldronConfigSchema = z.object({
	provider: z.string(),
	model: z.string(),
	apiKeyEnv: z.string(),
})

export type CauldronConfig = z.infer<typeof CauldronConfigSchema>
export type CauldronConfigPartial = z.infer<typeof CauldronConfigPartialSchema>
```

```typescript
/**
 * JSONC configuration loader for Cauldron.
 *
 * Reads global (~/.cauldron/config.jsonc) and project-local (.cauldron.jsonc) configs,
 * merges them (project overrides global), and validates the merged result with Zod.
 * Never throws — all errors are returned as Result<CauldronConfig, ZodError | Error>.
 */

import { readFile } from 'node:fs/promises'
import { homedir } from 'node:os'
import { join } from 'node:path'

import * as jsonc from 'jsonc-parser'
import type { ParseError } from 'jsonc-parser'
import * as z from 'zod'

import type { Result } from '@get-cauldron/types'
import { err, ok } from '@get-cauldron/types'

import type { CauldronConfig, CauldronConfigPartial } from './schema.js'
import { CauldronConfigSchema, CauldronConfigPartialSchema } from './schema.js'

async function loadFile(filePath: string): Promise<Result<CauldronConfigPartial, z.ZodError | Error>> {
	let text: string
	try {
		text = await readFile(filePath, 'utf8')
	} catch {
		// File doesn't exist — treat as empty partial, not an error
		return ok({})
	}

	const parseErrors: ParseError[] = []
	const raw = jsonc.parse(text, parseErrors, { allowTrailingComma: true })
	if (parseErrors.length > 0) {
		return err(new Error(`JSONC parse error in ${filePath}: ${parseErrors[0]!.error}`))
	}

	const result = CauldronConfigPartialSchema.safeParse(raw)
	if (!result.success) return err(result.error)
	return ok(result.data)
}

export async function loadConfig(
	projectDir: string,
	globalDir?: string,
): Promise<Result<CauldronConfig, z.ZodError | Error>> {
	const resolvedGlobalDir = globalDir ?? join(homedir(), '.cauldron')
	const globalPath = join(resolvedGlobalDir, 'config.jsonc')
	const projectPath = join(projectDir, '.cauldron.jsonc')

	const globalResult = await loadFile(globalPath)
	const projectResult = await loadFile(projectPath)

	if (!globalResult.ok) return globalResult
	if (!projectResult.ok) return projectResult

	const merged = { ...globalResult.value, ...projectResult.value }
	const validated = CauldronConfigSchema.safeParse(merged)
	if (!validated.success) return err(validated.error)
	return ok(validated.data)
}
```

| Boundary | Description |
|----------|-------------|
| filesystem → loader.ts | Config files are read from disk and parsed — content may be malformed but is developer-controlled (not user-supplied at runtime) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-P2-01 | Denial of Service | loadConfig / jsonc.parse | mitigate | jsonc.parse never throws; safeParse never throws; all errors returned as Result err() — malformed config cannot crash the process |
| T-02-P2-02 | Tampering | config path construction | accept | Paths constructed from `os.homedir()` + hardcoded filenames (`config.jsonc`, `.cauldron.jsonc`) — no user-supplied path segments; no path traversal risk |
| T-02-P2-03 | Information Disclosure | apiKeyEnv config field | accept | Field stores env var NAME (e.g., "ANTHROPIC_API_KEY"), not the key value — key stays in environment, never in config file, never in CauldronConfig object |
| T-02-P2-04 | Tampering | ZodError in Result | accept | ZodError is typed; callers receive structured error; no sensitive data in Zod validation errors for this schema (provider/model/apiKeyEnv are non-sensitive strings) |
