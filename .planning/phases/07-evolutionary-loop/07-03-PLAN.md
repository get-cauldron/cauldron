    ```typescript
    import { z } from 'zod';
    import type { LLMGateway } from '../gateway/gateway.js';
    import type { Seed } from '@cauldron/shared';
    import type { GapAnalysis, LateralThinkingProposal } from './types.js';

    export const PERSONAS = ['contrarian', 'hacker', 'simplifier', 'researcher', 'architect'] as const;
    export type Persona = typeof PERSONAS[number];
    ```

    **PERSONA_PROMPTS**: A `Record<Persona, string>` mapping each persona to its system-level instruction:
    - `contrarian`: "Challenge every assumption in the current approach. Propose the opposite of what failed. If the current spec is complex, propose radical simplicity. If it's simple, propose depth."
    - `hacker`: "Find the fastest, most pragmatic path to the goal. Cut unnecessary abstractions. Propose the minimal viable approach that would actually work."
    - `simplifier`: "Remove complexity. Merge overlapping criteria. Identify what can be eliminated entirely. The best code is no code."
    - `researcher`: "What proven patterns or libraries solve this problem? Propose an approach grounded in established solutions, not invention."
    - `architect`: "Rethink the structural decomposition. Maybe the molecule/bead boundaries are wrong. Propose a different way to organize the work."

    **ProposalSchema** (LLM-safe, no min/max):
    ```typescript
    const ProposalSchema = z.object({
      goal: z.string(),
      constraints: z.array(z.object({ constraint: z.string() })),
      acceptanceCriteria: z.array(z.object({ criterion: z.string() })),
      rationale: z.string(),
    });
    ```

    **MetaJudgeSchema**:
    ```typescript
    const MetaJudgeSchema = z.object({
      selectedPersona: z.string().nullable(),
      mergedProposal: ProposalSchema.nullable(),
      reasoning: z.string(),
      viable: z.boolean(),
    });
    ```

    **generatePersonaProposal(params: { gateway, persona, seed, gapAnalysis, projectId, seedId })**: Build prompt from PERSONA_PROMPTS[persona] + seed goal + current ACs + gap analysis. Call `gateway.generateObject` with ProposalSchema at stage 'evaluation'. Return `LateralThinkingProposal` with persona field set.

    **metaJudgeSelect(params: { gateway, proposals, originalSeed, projectId, seedId })**: Build prompt listing all 5 proposals with their persona labels and rationales. Ask the meta-judge to select the most promising or merge complementary ideas. Call `gateway.generateObject` with MetaJudgeSchema. If `viable === false`, return null. Otherwise, return the merged/selected proposal mapped to a SeedSummary-compatible shape.

    **runLateralThinking(params: { step, gateway, seed, gapAnalysis, projectId, seedId })**:
    Per D-14 and D-15:
    ```typescript
    export async function runLateralThinking(params: {
      step: { run: <T>(name: string, fn: () => Promise<T>) => Promise<T> };
      gateway: LLMGateway;
      seed: Seed;
      gapAnalysis: GapAnalysis[];
      projectId: string;
      seedId: string;
    }): Promise<LateralThinkingProposal | null> {
      // Run all 5 personas in parallel via step.run
      const proposals = await Promise.all(
        PERSONAS.map(persona =>
          params.step.run(`lateral-thinking-${persona}`, () =>
            generatePersonaProposal({
              gateway: params.gateway,
              persona,
              seed: params.seed,
              gapAnalysis: params.gapAnalysis,
              projectId: params.projectId,
              seedId: params.seedId,
            })
          )
        )
      );

      // Meta-judge selects the best proposal
      const selected = await params.step.run('lateral-thinking-meta-judge', () =>
        metaJudgeSelect({
          gateway: params.gateway,
          proposals,
          originalSeed: params.seed,
          projectId: params.projectId,
          seedId: params.seedId,
        })
      );

      return selected;
    }
    ```
