**docker-compose.yml** (per Pattern 6 from RESEARCH.md):
```yaml
services:
  postgres:
    image: postgres:17-alpine
    environment:
      POSTGRES_USER: cauldron
      POSTGRES_PASSWORD: cauldron
      POSTGRES_DB: cauldron
    ports:
      - '5432:5432'
    volumes:
      - postgres_data:/var/lib/postgresql/data
    healthcheck:
      test: ['CMD-SHELL', 'pg_isready -U cauldron']
      interval: 5s
      timeout: 5s
      retries: 5

  postgres-test:
    image: postgres:17-alpine
    environment:
      POSTGRES_USER: cauldron
      POSTGRES_PASSWORD: cauldron
      POSTGRES_DB: cauldron_test
    ports:
      - '5433:5432'
    healthcheck:
      test: ['CMD-SHELL', 'pg_isready -U cauldron']
      interval: 5s
      timeout: 5s
      retries: 5

  redis:
    image: redis:7-alpine
    ports:
      - '6379:6379'
    healthcheck:
      test: ['CMD', 'redis-cli', 'ping']
      interval: 5s
      timeout: 5s
      retries: 5

  inngest:
    image: inngest/inngest:latest
    command: 'inngest dev -u http://host.docker.internal:3001/api/inngest'
    ports:
      - '8288:8288'
      - '8289:8289'
    depends_on:
      redis:
        condition: service_healthy
    environment:
      - INNGEST_DEV=1

volumes:
  postgres_data:
```

Note: Two Postgres instances — one for dev (port 5432), one for integration tests (port 5433, database `cauldron_test`). This avoids test pollution of dev data.

The Inngest `command` uses `host.docker.internal` because the engine process runs on the host during development, not inside Docker. This is the standard Docker-for-Mac/Linux pattern for reaching host services.

**scripts/wait-for-services.sh** — A utility to wait for Docker services to be healthy before running tests:
```bash
#!/usr/bin/env bash
set -euo pipefail

echo "Waiting for PostgreSQL..."
until docker compose exec -T postgres pg_isready -U cauldron > /dev/null 2>&1; do
  sleep 1
done
echo "PostgreSQL is ready."

echo "Waiting for PostgreSQL (test)..."
until docker compose exec -T postgres-test pg_isready -U cauldron > /dev/null 2>&1; do
  sleep 1
done
echo "PostgreSQL (test) is ready."

echo "Waiting for Redis..."
until docker compose exec -T redis redis-cli ping > /dev/null 2>&1; do
  sleep 1
done
echo "Redis is ready."

echo "All services ready."
```

Make the script executable: `chmod +x scripts/wait-for-services.sh`
