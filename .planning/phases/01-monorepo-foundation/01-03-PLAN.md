    ```typescript
    /**
     * Fixture recording/replay infrastructure for deterministic LLM response testing.
     *
     * Phase 1: Type definitions and helper utilities only.
     * Phase 3+: Actual fixture recording against live providers.
     */

    /** A single recorded LLM response chunk (text or SSE event). */
    export interface FixtureChunk {
      /** Milliseconds since recording start */
      timestamp: number;
      /** Raw chunk data (string for text, object for structured events) */
      data: string;
    }

    /** A complete recorded LLM interaction. */
    export interface FixtureRecording {
      /** Provider name (e.g., "anthropic", "openai") */
      provider: string;
      /** Model identifier used in the recording */
      model: string;
      /** ISO 8601 timestamp of when this was recorded */
      recordedAt: string;
      /** The request that produced this response */
      request: {
        messages: Array<{ role: string; content: string }>;
        maxTokens?: number;
      };
      /** Ordered sequence of response chunks */
      chunks: FixtureChunk[];
    }

    /** Options for replaying a fixture recording. */
    export interface FixtureReplayOptions {
      /** Whether to simulate original timing delays between chunks */
      simulateLatency: boolean;
    }

    /**
     * Load a fixture recording from a JSON file path.
     * Returns the parsed FixtureRecording or throws if file is malformed.
     */
    export async function loadFixture(filePath: string): Promise<FixtureRecording> {
      const { readFile } = await import("node:fs/promises");
      const raw = await readFile(filePath, "utf-8");
      const parsed: unknown = JSON.parse(raw);

      if (!isFixtureRecording(parsed)) {
        throw new Error(`Invalid fixture format at ${filePath}`);
      }

      return parsed;
    }

    /**
     * Type guard for FixtureRecording.
     */
    export function isFixtureRecording(value: unknown): value is FixtureRecording {
      if (typeof value !== "object" || value === null) return false;
      const obj = value as Record<string, unknown>;
      return (
        typeof obj.provider === "string" &&
        typeof obj.model === "string" &&
        typeof obj.recordedAt === "string" &&
        Array.isArray(obj.chunks)
      );
    }
    ```
