This is a unit test (no DB needed) that creates a tRPC caller with `authenticated: false` and verifies mutations reject while queries pass through. Use the `appRouter.createCaller()` pattern.

```typescript
import { describe, it, expect } from 'vitest';
import { appRouter } from '../../router.js';

// Minimal mock DB — mutations should reject BEFORE hitting DB
const mockDb = new Proxy({} as any, {
  get: () => () => { throw new Error('DB should not be called for auth rejection'); },
});

// Unauthenticated caller — simulates a request with no/invalid API key
const unauthCaller = appRouter.createCaller({
  db: mockDb,
  authenticated: false,
  getEngineDeps: async () => { throw new Error('should not be called'); },
});

describe('Auth Middleware — SEC-02', () => {
  describe('mutations reject when unauthenticated', () => {
    it('projects.create rejects with UNAUTHORIZED', async () => {
      await expect(
        unauthCaller.projects.create({ name: 'blocked' })
      ).rejects.toMatchObject({ code: 'UNAUTHORIZED' });
    });

    it('projects.archive rejects with UNAUTHORIZED', async () => {
      await expect(
        unauthCaller.projects.archive({ id: '00000000-0000-0000-0000-000000000000' })
      ).rejects.toMatchObject({ code: 'UNAUTHORIZED' });
    });

    it('projects.delete rejects with UNAUTHORIZED', async () => {
      await expect(
        unauthCaller.projects.delete({ id: '00000000-0000-0000-0000-000000000000' })
      ).rejects.toMatchObject({ code: 'UNAUTHORIZED' });
    });

    it('projects.updateSettings rejects with UNAUTHORIZED', async () => {
      await expect(
        unauthCaller.projects.updateSettings({
          id: '00000000-0000-0000-0000-000000000000',
          settings: {},
        })
      ).rejects.toMatchObject({ code: 'UNAUTHORIZED' });
    });

    it('interview.startInterview rejects with UNAUTHORIZED', async () => {
      await expect(
        unauthCaller.interview.startInterview({ projectId: 'test' })
      ).rejects.toMatchObject({ code: 'UNAUTHORIZED' });
    });

    it('interview.sendAnswer rejects with UNAUTHORIZED', async () => {
      await expect(
        unauthCaller.interview.sendAnswer({ projectId: 'test', answer: 'test' })
      ).rejects.toMatchObject({ code: 'UNAUTHORIZED' });
    });

    it('interview.approveSummary rejects with UNAUTHORIZED', async () => {
      await expect(
        unauthCaller.interview.approveSummary({
          projectId: 'test',
          summary: {
            goal: 'test',
            constraints: [],
            acceptanceCriteria: [],
            ontologySchema: { entities: [] },
            evaluationPrinciples: [],
            exitConditions: [],
          },
        })
      ).rejects.toMatchObject({ code: 'UNAUTHORIZED' });
    });

    it('execution.triggerDecomposition rejects with UNAUTHORIZED', async () => {
      await expect(
        unauthCaller.execution.triggerDecomposition({
          projectId: '00000000-0000-0000-0000-000000000000',
          seedId: '00000000-0000-0000-0000-000000000000',
        })
      ).rejects.toMatchObject({ code: 'UNAUTHORIZED' });
    });

    it('execution.triggerExecution rejects with UNAUTHORIZED', async () => {
      await expect(
        unauthCaller.execution.triggerExecution({
          projectId: '00000000-0000-0000-0000-000000000000',
          seedId: '00000000-0000-0000-0000-000000000000',
        })
      ).rejects.toMatchObject({ code: 'UNAUTHORIZED' });
    });

    it('execution.respondToEscalation rejects with UNAUTHORIZED', async () => {
      await expect(
        unauthCaller.execution.respondToEscalation({
          projectId: '00000000-0000-0000-0000-000000000000',
          action: 'retry',
        })
      ).rejects.toMatchObject({ code: 'UNAUTHORIZED' });
    });
  });

  describe('queries pass through when unauthenticated', () => {
    // Queries hit the DB, so we need a minimal mock that returns empty results
    const queryDb = new Proxy({} as any, {
      get: (_target, prop) => {
        // Drizzle query builder chain: db.select().from().where().orderBy().limit()
        const chainable = () => new Proxy({} as any, {
          get: () => chainable,
          apply: () => new Proxy([], {
            get: (target, p) => p in target ? (target as any)[p] : chainable,
          }),
        });
        if (prop === 'execute') return async () => [];
        return chainable;
      },
    });

    const queryUnauthCaller = appRouter.createCaller({
      db: queryDb,
      authenticated: false,
      getEngineDeps: async () => { throw new Error('should not be called'); },
    });

    it('projects.list succeeds when unauthenticated', async () => {
      // Should not throw UNAUTHORIZED — queries are public
      await expect(
        queryUnauthCaller.projects.list()
      ).resolves.not.toThrow();
    });
  });
});
```
