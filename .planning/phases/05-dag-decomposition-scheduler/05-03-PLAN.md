    ```typescript
    // Use Drizzle's sql template for the NOT EXISTS subquery
    const readyBeads = await db
      .select()
      .from(beads)
      .where(
        and(
          eq(beads.seedId, seedId),
          eq(beads.status, 'pending'),
          sql`NOT EXISTS (
            SELECT 1 FROM bead_edges
            INNER JOIN beads AS blockers ON bead_edges.from_bead_id = blockers.id

      );
    return readyBeads;
    ```

    3. Step "check-upstream-waits": Query beadEdges for `waits_for` edges where `toBeadId = beadId`. If any exist, use `Promise.all()` with `step.waitForEvent()` for each upstream (per Research Pattern 2):
       ```typescript
       const waitsForEdges = await step.run('check-upstream-waits', async () => {
         // Query edges from DB
         return db.select().from(beadEdges).where(and(eq(beadEdges.toBeadId, beadId), eq(beadEdges.edgeType, 'waits_for')));
       });
       if (waitsForEdges.length > 0) {
         const results = await Promise.all(
           waitsForEdges.map(edge =>
             step.waitForEvent(`wait-for-bead-${edge.fromBeadId}`, {
               event: 'bead.completed',
               match: 'data.beadId',
               timeout: '2h',
             })
           )
         );
         const timedOut = results.some(r => r === null);
         if (timedOut) {
           // Mark as failed due to timeout
           await step.run('mark-timeout-failed', async () => {
             await completeBead(db, beadId, 'failed', projectId, seedId);
           });
           return { beadId, status: 'failed' };
         }
       }
       ```

       ```typescript
       const conditionalEdge = await step.run('check-conditional', async () => {
         const edges = await db.select().from(beadEdges).where(and(eq(beadEdges.toBeadId, beadId), eq(beadEdges.edgeType, 'conditional_blocks')));
         if (edges.length === 0) return null;
         const [upstream] = await db.select().from(beads).where(eq(beads.id, edges[0].fromBeadId));
         return upstream;
       });
       if (conditionalEdge && conditionalEdge.status === 'failed') {
         await step.run('skip-conditional', async () => {
       ```

    ```typescript
      tokenBudget?: number;

      beadDbIds: Map<string, string>;

    }

    /**
     * Full decomposition pipeline: decompose seed -> persist to DB -> dispatch all ready beads.
     * This is the single entry point Phase 6 calls after seed crystallization.
     */
    export async function runDecomposition(options: RunDecompositionOptions): Promise<RunDecompositionResult> {
      const { db, gateway, inngest, seed, projectId, maxRetries, tokenBudget } = options;

      // 1. Emit decomposition_started event
      await appendEvent(db, {
        projectId,
        seedId: seed.id,
        type: 'decomposition_started',
        payload: { seedId: seed.id },
      });

      try {
        // 2. Run two-pass LLM decomposition with validation and retry
        const decomposition = await decomposeSeed({
          gateway,
          seed,
          projectId,
          maxRetries,
          tokenBudget,
        });

        // 3. Persist molecules, beads, and edges to database
        const { moleculeDbIds, beadDbIds } = await persistDecomposition(db, seed.id, decomposition);

        // 4. Emit decomposition_completed event
        await appendEvent(db, {
          projectId,
          seedId: seed.id,
          type: 'decomposition_completed',
          payload: { moleculeCount: decomposition.molecules.length, beadCount: decomposition.beads.length },
        });

        // 5. Find all immediately ready beads (no blocking dependencies) per D-12
        const readyBeads = await findReadyBeads(db, seed.id);

        // 6. Dispatch all ready beads via Inngest events
        const dispatchedBeadIds: string[] = [];
        for (const bead of readyBeads) {
          const payload: BeadDispatchPayload = {
            beadId: bead.id,
            seedId: seed.id,
            projectId,
            moleculeId: bead.moleculeId,
          };
          await inngest.send({ name: 'bead.dispatch_requested', data: payload });
          dispatchedBeadIds.push(bead.id);
        }

        return { decomposition, moleculeDbIds, beadDbIds, dispatchedBeadIds };
      } catch (error) {
        // Emit decomposition_failed event on any error
        await appendEvent(db, {
          projectId,
          seedId: seed.id,
          type: 'decomposition_failed',
          payload: { error: error instanceof Error ? error.message : String(error) },
        });
        throw error;
      }
    }
    ```
