   ```typescript
   export async function crystallizeSeed(
     db: DbClient,
     interviewId: string,
     projectId: string,
     summary: SeedSummary,
     ambiguityScore: number,
     parentSeedId?: string,
   ): Promise<Seed> {
     // Check if a crystallized seed already exists for this interview
     const existing = await db.select().from(seeds).where(eq(seeds.interviewId, interviewId));
     const crystallized = existing.find((s) => s.status === 'crystallized');
     if (crystallized) {
       throw new ImmutableSeedError(crystallized.id);
     }

     // Compute version: parent's version + 1, or 1 if no parent
     let version = 1;
     if (parentSeedId) {
       const [parent] = await db.select().from(seeds).where(eq(seeds.id, parentSeedId));
       if (parent) version = parent.version + 1;
     }

     // INSERT new seed as crystallized (D-25: never UPDATE, always INSERT)
     const [seed] = await db.insert(seeds).values({
       projectId,
       interviewId,
       parentId: parentSeedId ?? null,
       version,
       status: 'crystallized',
       goal: summary.goal,
       constraints: summary.constraints,
       acceptanceCriteria: summary.acceptanceCriteria,
       ontologySchema: summary.ontologySchema,
       evaluationPrinciples: summary.evaluationPrinciples,
       exitConditions: summary.exitConditions,
       ambiguityScore,
       crystallizedAt: new Date(),
     }).returning();

     // Update interview status to 'completed' and phase to 'crystallized'
     await db.update(interviews)
       .set({ status: 'completed', phase: 'crystallized', completedAt: new Date() })
       .where(eq(interviews.id, interviewId));

     // Append seed_crystallized event to event store
     await appendEvent(db, {
       projectId,
       seedId: seed!.id,
       type: 'seed_crystallized',
       payload: { interviewId, ambiguityScore, version, parentSeedId: parentSeedId ?? null },
     });

     return seed!;
   }
   ```

   **Seed lineage query (D-27, SEED-04):**
   ```typescript
   export async function getSeedLineage(
     db: DbClient,
     seedId: string,
   ): Promise<Seed[]> {
     const result = await db.execute(sql`
       WITH RECURSIVE lineage AS (
         SELECT * FROM seeds WHERE id = ${seedId}::uuid
         UNION ALL
         SELECT s.* FROM seeds s INNER JOIN lineage l ON s.id = l.parent_id
       )
       SELECT * FROM lineage ORDER BY version ASC
     `);
     return result.rows as unknown as Seed[];
   }
   ```

   **InterviewFSM class (D-02, D-03, D-05, D-06):**
   ```typescript
   export class InterviewFSM {
     constructor(
       private readonly db: DbClient,
       private readonly gateway: LLMGateway,
       private readonly config: GatewayConfig,
       private readonly logger: Logger,
     ) {}

     // Start a new interview or resume an existing one (D-03)
     async startOrResume(
       projectId: string,
       mode?: InterviewMode,
     ): Promise<Interview> {
       // Check for existing active/paused interview
       const [existing] = await this.db.select().from(interviews)
         .where(eq(interviews.projectId, projectId))
         .orderBy(desc(interviews.createdAt))
         .limit(1);

       if (existing && (existing.status === 'active' || existing.status === 'paused')) {
         // Resume: set status to active if paused
         if (existing.status === 'paused') {
           await this.db.update(interviews)
             .set({ status: 'active' })
             .where(eq(interviews.id, existing.id));
         }
         return { ...existing, status: 'active' };
       }

       // Create new interview
       const [interview] = await this.db.insert(interviews).values({
         projectId,
         mode: mode ?? 'greenfield',
         status: 'active',
         phase: 'gathering',
       }).returning();

       await appendEvent(this.db, {
         projectId,
         type: 'interview_started',
         payload: { interviewId: interview!.id, mode: mode ?? 'greenfield' },
       });

       return interview!;
     }

     // Submit an answer and get the next question (D-07, D-21)
     async submitAnswer(
       interviewId: string,
       answer: string,
       freeformText?: string,
     ): Promise<TurnResult> {
       const [interview] = await this.db.select().from(interviews)
         .where(eq(interviews.id, interviewId));

       if (!interview) throw new Error(`Interview ${interviewId} not found`);
       if (interview.status !== 'active') throw new Error(`Interview ${interviewId} is ${interview.status}, not active`);
       if (interview.phase !== 'gathering') throw new Error(`Interview ${interviewId} is in phase ${interview.phase}, not gathering`);

       const transcript = (interview.transcript ?? []) as InterviewTurn[];
       const previousScores = interview.currentAmbiguityScore as AmbiguityScores | null;
       const turnCount = interview.turnCount;

       // D-21: Run scoring and perspective calls in parallel
       const [scores, candidates] = await Promise.all([
         scoreTranscript(
           this.gateway, transcript, interview.mode as InterviewMode,
           interview.projectId, previousScores, { scoringModel: this.config.scoringModel },
         ),
         runActivePerspectives(
           this.gateway, transcript, previousScores, interview.projectId,
           turnCount, { perspectiveModels: this.config.perspectiveModels },
         ),
       ]);

       // D-04/D-05: Check threshold
       const thresholdMet = scores.overall >= 0.8; // ambiguity <= 0.2 means clarity >= 0.8

       // Rank candidates and get next question (unless threshold met)
       let nextQuestion: RankedQuestion | null = null;
       if (!thresholdMet) {
         nextQuestion = await rankCandidates(this.gateway, candidates, transcript, interview.projectId);
       }

       // Build the turn record (D-07)
       const turn: InterviewTurn = {
         turnNumber: turnCount + 1,
         perspective: nextQuestion?.selectedCandidate.perspective ?? candidates[0]?.perspective ?? 'researcher',
         question: nextQuestion?.selectedCandidate.question ?? 'Interview complete — ready for review.',
         mcOptions: nextQuestion?.mcOptions ?? [],
         userAnswer: answer,
         freeformText,
         ambiguityScoreSnapshot: scores,
         model: nextQuestion?.selectedCandidate.model ?? 'system',
         allCandidates: candidates,
         timestamp: new Date().toISOString(),
       };

       // Persist turn to transcript
       const updatedTranscript = [...transcript, turn];
       const scoresHistory = [...((interview.ambiguityScoresHistory ?? []) as AmbiguityScores[]), scores];

       const updateData: Record<string, unknown> = {
         transcript: updatedTranscript,
         ambiguityScoresHistory: scoresHistory,
         currentAmbiguityScore: scores,
         turnCount: turnCount + 1,
       };

       // Auto-transition to reviewing when threshold met (D-02)
       if (thresholdMet) {
         updateData.phase = 'reviewing';
       }

       await this.db.update(interviews).set(updateData).where(eq(interviews.id, interviewId));

       return { turn, scores, nextQuestion, thresholdMet };
     }

     // Request early crystallization warning (D-06)
     async requestEarlyCrystallization(interviewId: string): Promise<EarlyCrystallizationWarning> {
       const [interview] = await this.db.select().from(interviews)
         .where(eq(interviews.id, interviewId));

       if (!interview) throw new Error(`Interview ${interviewId} not found`);

       const scores = interview.currentAmbiguityScore as AmbiguityScores | null;
       const currentScore = scores?.overall ?? 1.0;
       const threshold = 0.8; // clarity threshold (1 - 0.2 ambiguity threshold)

       const dimensions = [
         { dimension: 'goalClarity', score: scores?.goalClarity ?? 0 },
         { dimension: 'constraintClarity', score: scores?.constraintClarity ?? 0 },
         { dimension: 'successCriteriaClarity', score: scores?.successCriteriaClarity ?? 0 },
       ];
       if (scores?.contextClarity !== undefined) {
         dimensions.push({ dimension: 'contextClarity', score: scores.contextClarity });
       }

       const weakest = dimensions.sort((a, b) => a.score - b.score).slice(0, 2);

       // Transition to reviewing for early crystallization
       await this.db.update(interviews)
         .set({ phase: 'reviewing' })
         .where(eq(interviews.id, interviewId));

       return {
         currentScore,
         threshold,
         gap: threshold - currentScore,
         weakestDimensions: weakest,
         message: `Warning: Current clarity score is ${(currentScore * 100).toFixed(0)}% (threshold: ${(threshold * 100).toFixed(0)}%). Weakest: ${weakest.map((w) => `${w.dimension}: ${(w.score * 100).toFixed(0)}%`).join(', ')}. Proceeding will create an underspecified seed.`,
       };
     }

     // Generate summary for review (D-22, INTV-06)
     async generateSummary(interviewId: string): Promise<SeedSummary> {
       const [interview] = await this.db.select().from(interviews)
         .where(eq(interviews.id, interviewId));

       if (!interview) throw new Error(`Interview ${interviewId} not found`);
       if (interview.phase !== 'reviewing') throw new Error(`Interview must be in reviewing phase, currently: ${interview.phase}`);

       const transcript = (interview.transcript ?? []) as InterviewTurn[];
       return synthesizeFromTranscript(this.gateway, transcript, interview.projectId);
     }

     // Approve summary and crystallize (D-23, INTV-07, SEED-01, SEED-02)
     async approveAndCrystallize(
       interviewId: string,
       summary: SeedSummary,
       parentSeedId?: string,
     ): Promise<Seed> {
       const [interview] = await this.db.select().from(interviews)
         .where(eq(interviews.id, interviewId));

       if (!interview) throw new Error(`Interview ${interviewId} not found`);
       if (interview.phase !== 'reviewing' && interview.phase !== 'approved') {
         throw new Error(`Interview must be in reviewing or approved phase, currently: ${interview.phase}`);
       }

       // Transition to approved
       assertValidTransition(interview.phase as InterviewPhase, 'approved');
       await this.db.update(interviews)
         .set({ phase: 'approved' })
         .where(eq(interviews.id, interviewId));

       const scores = interview.currentAmbiguityScore as AmbiguityScores | null;
       const ambiguityScore = scores ? 1 - scores.overall : 1.0; // Store as ambiguity (inverse of clarity)

       // Crystallize the seed
       const seed = await crystallizeSeed(
         this.db, interviewId, interview.projectId,
         summary, ambiguityScore, parentSeedId,
       );

       this.logger.info({ seedId: seed.id, interviewId, ambiguityScore }, 'Seed crystallized');

       return seed;
     }

     // Pause interview (D-03)
     async pause(interviewId: string): Promise<void> {
       const [interview] = await this.db.select().from(interviews)
         .where(eq(interviews.id, interviewId));

       if (!interview) throw new Error(`Interview ${interviewId} not found`);
       if (interview.status !== 'active') throw new Error(`Can only pause active interviews`);

       await this.db.update(interviews)
         .set({ status: 'paused' })
         .where(eq(interviews.id, interviewId));
     }

     // Abandon interview
     async abandon(interviewId: string): Promise<void> {
       const [interview] = await this.db.select().from(interviews)
         .where(eq(interviews.id, interviewId));
   ```
