    Strategy: Use raw SQL via Drizzle's `sql` template tag to build a single query with lateral subqueries (PostgreSQL LATERAL JOIN). This is cleaner than multiple LEFT JOINs with window functions for this use case.

    Replace lines 16-58 of the `list` procedure with:

    ```typescript
    const rows = await ctx.db.execute(sql`
      SELECT
        p.id, p.name, p.description, p.settings,
        p.created_at AS "createdAt",
        p.updated_at AS "updatedAt",
        COALESCE(le.occurred_at, p.created_at) AS "lastActivity",
        le.type AS "lastEventType",
        COALESCE(cu.total_cost, 0)::numeric AS "totalCostCents"
      FROM projects p
      LEFT JOIN LATERAL (
        SELECT e.type, e.occurred_at
        FROM events e
        WHERE e.project_id = p.id
        ORDER BY e.occurred_at DESC
        LIMIT 1
      ) le ON true
      LEFT JOIN LATERAL (
        SELECT SUM(u.cost_cents) AS total_cost
        FROM llm_usage u
        WHERE u.project_id = p.id
      ) cu ON true
      WHERE p.deleted_at IS NULL
        ${includeArchived ? sql`` : sql`AND p.name NOT LIKE '[archived]%'`}
      ORDER BY p.updated_at DESC
    `);
    ```
